204  template<typename T>
205  double array_stdev(const T data[], const int array_size) {
206  double sum = 0;
207  double avg = array_avg(data, array_size);
208  for (int i = 0; i < array_size; i++) {
209  sum += (data[i] - avg)*(data[i] - avg);
210  }
211  return sqrt(sum / (array_size - 1));
212  }

218  template<typename base_type>
219  base_type intpow(const base_type base, const int exponent) {
220  base_type result;
221  if (exponent == 0) {
222  result = 1;
223  return result;
224  }
225  result = base;
226  for (int i = 1; i < abs(exponent); i++) {

228  }
229  if (exponent < 0) {
230  result = 1.0 / result;

266  template<typename T>
267  void removeDuplicates(std::vector<T> &vec) {
268  if ((int)vec.size() > 2) {
269  for (auto it1 = vec.begin(); it1 != vec.end() - 2; ++it1) {
270  for (auto it2 = it1 + 2; it2 != vec.end(); ++it2) {
271  if (*it2 == *it1) {
272  *it2 = *(it2 - 1);
273  }
274  }
275  }
276  }
277  if ((int)vec.size() > 1) {
278  auto it3 = unique(vec.begin(), vec.end());
279  vec.resize(distance(vec.begin(), it3));

295  double vector_stdev(const std::vector<T, A>& data) {
296  double avg = vector_avg(data);
297  double sum = accumulate(data.begin(), data.end(), 0.0, [avg](double& x, const T& element) {
298  return x + (element - avg)*(element - avg);
