314  template<typename T>
315  double intpow(const T base, const int exponent) {
316  double result;
317  if (exponent == 0) {
318  result = 1.0;
319  return result;
320  }
321  result = base;
322  for (int i = 1; i < abs(exponent); i++) {
323  result *= base;
324  }
325  if (exponent < 0) {
326  result = 1.0 / result;
327  }
328  return result;

362  template<typename T>
363  void removeDuplicates(std::vector<T> &vec) {
364  if ((int)vec.size() > 2) {
365  for (auto it1 = vec.begin(); it1 != vec.end() - 2; ++it1) {
366  for (auto it2 = it1 + 2; it2 != vec.end(); ++it2) {
367  if (*it2 == *it1) {
368  *it2 = *(it2 - 1);
369  }
370  }
371  }
372  }
373  if ((int)vec.size() > 1) {
374  auto it3 = unique(vec.begin(), vec.end());
375  vec.resize(distance(vec.begin(), it3));
376  }

394  template<typename T, typename A>
395  double vector_median(std::vector<T, A> data) {
396  if (data.size() % 2 == 0) {
397  const auto median_it1 = data.begin() + data.size() / 2 - 1;
398  const auto median_it2 = data.begin() + data.size() / 2;
399  std::nth_element(data.begin(), median_it1, data.end());
400  T val1 = *median_it1;
401  std::nth_element(data.begin(), median_it2, data.end());
402  T val2 = *median_it2;
403  return (double)(val1 + val2) / 2.0;
404  }
405  else {
406  const auto median_it = data.begin() + data.size() / 2;
407  std::nth_element(data.begin(), median_it, data.end());
408  return *median_it;
409  }
410  }
